Algoritmo booleanos
	a <- 5 == 3
	// a = Falso
	// Explicación: Se está realizando una comparación entre 5 es igual a 3, por lo que es Falso, ya que no pueden ser iguales.
	b <- 4 <> 3
	// b = Verdadera

	//Explicación:Vemos que se realiza una comparación de que el número 4 es menor a 4, por lo que es falso ya que es el número mismo y no puede ser menor.
